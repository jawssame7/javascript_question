## 下記の関数を実行したときどういった結果が帰ってくるかを回答してください。
`word()` の引数は `Cheese` とする


``` javascript
function word(w) {
    var arr = [];

    for(var i = 0; i < 10; i++) {
        if (i === 0) {
            arr.push('A');
        } else if (i === 1 || i === 7) {
            arr.push(' ');
        } else if (i === 2) {
            arr.push('p');
        } else if (i === 3) {
            arr.push('i');
        } else if (i === 4 || i === 6) {
            arr.push('e');
        } else if (i === 5) {
            arr.push('c');
        } else if (i === 8) {
            arr.push('o');
        } else if (i === 9) {
            arr.push('f');
        }
    }
    arr.push(' ');
    arr.push(w);
    return arr.join('');
}


```
